create database projetphp;
create table projetphp.admin(
username varchar(255) primary key,
password varchar(255)
);
create table projetphp.doctor(
nom varchar(255),
prenom varchar(255),
CIN varchar(255) primary key,
date_naissance date,
ID_med varchar(255),
tel varchar(255),
E_mail varchar(255),
Hopital varchar(255),
service varchar(255),
password varchar(255),
accepted varchar(255)
);
create table projetphp.visiteur(
nom varchar(255),
prenom varchar(255),
tel varchar(255),
E_mail varchar(255) primary key,
password varchar(255)
);
